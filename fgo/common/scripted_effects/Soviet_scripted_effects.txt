

SOV_increase_zinoviev = {
	##提高季诺维也夫影响力
	custom_effect_tooltip = SOV_increase_zinoviev_tooltip
	hidden_effect = {
		if = {
			limit = { has_idea = SOV_kamenev_prevail_3 }
			swap_ideas = {
				remove_idea = SOV_kamenev_prevail_3
				add_idea = SOV_kamenev_prevail_2
			}
		}
		else_if = {
			limit = { has_idea = SOV_kamenev_prevail_2 }
			swap_ideas = {
				remove_idea = SOV_kamenev_prevail_2
				add_idea = SOV_kamenev_prevail_1
			}
		}
		else_if = {
			limit = { has_idea = SOV_kamenev_prevail_1 }
			swap_ideas = {
				remove_idea = SOV_kamenev_prevail_1
				add_idea = SOV_zinoviev_kamenev_0
			}
		}
		else_if = {
			limit = { has_idea = SOV_zinoviev_kamenev_0 }
			swap_ideas = {
				remove_idea = SOV_zinoviev_kamenev_0
				add_idea = SOV_zinoviev_prevail_1
			}
		}
		else_if = {
			limit = { has_idea = SOV_zinoviev_prevail_1 }
			swap_ideas = {
				remove_idea = SOV_zinoviev_prevail_1
				add_idea = SOV_zinoviev_prevail_2
			}
		}
		else_if = {
			limit = { has_idea = SOV_zinoviev_prevail_2 }
			swap_ideas = {
				remove_idea = SOV_zinoviev_prevail_2
				add_idea = SOV_zinoviev_prevail_3
			}
		}
		else = {
			country_event = {
				##季诺维也夫过多
				id = SOV_F.33
				days = 3
				random = 72
			}
		}
		if = {
			limit = {
				OR = {
					has_idea = SOV_zinoviev_prevail_1
					has_idea = SOV_zinoviev_prevail_2
					has_idea = SOV_zinoviev_prevail_3
				}
				has_country_leader = {
					name = "Lev Kamenev"
					ruling_only = yes
				}
			}
			country_event = {
				##季诺维也夫上
				id = SOV_F.32
				days = 3
				random = 72
			}
		}
	}
}
